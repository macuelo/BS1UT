import java.util.ArrayList;

public class Main {
    public static void main(String[] args) {
        // Crear lista de juegos
        ArrayList<Juego> juegos = new ArrayList<>();

        // Agregar juegos a la lista
        juegos.add(new Juego(1, "Minecraft", "Sandbox", 500));
        juegos.add(new Juego(2, "The Witcher 3", "RPG", 200));
        juegos.add(new Juego(3, "FIFA 23", "Deportes", 300));

        // Buscar juegos de g√©nero RPG
        for (Juego j : juegos) {
            if (j.getGenero().equalsIgnoreCase("RPG")) {
                System.out.println("Encontrado: " + j.getTitulo());
            }
        }
    }
}
